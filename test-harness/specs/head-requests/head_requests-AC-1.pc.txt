{
  "info": {
    "_postman_id": "37c4cd69-28e5-48aa-8386-a4ec010216e4"
  },
  "item": [
    {

      "description": "todos-head",
      "request": {
        "url": {
          "path": ["todos"]
        },
        "method": "HEAD"
      },
      "response": [
        {
          "description": "todos-head-200",
          "code": 200,
          "header": [
            {
              "key": "nonce",
              "value": "abc"
            }
          ]
        }
      ]
    }
  ]
}

mock -p 4010 ${document}
